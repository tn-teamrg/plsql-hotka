
set autocommit off;
DECLARE
   TYPE DEPTNO_TYPE IS TABLE OF DEPT.DEPTNO%TYPE;
   DEPTNO_TABLE DEPTNO_TYPE;
BEGIN
   DELETE from DEPT
   RETURNING DEPTNO BULK COLLECT INTO DEPTNO_TABLE;
   DBMS_OUTPUT.PUT_LINE(DEPTNO_TABLE.COUNT || ' would be deleted');

   ROLLBACK;
   
   FOR i in DEPTNO_TABLE.FIRST .. DEPTNO_TABLE.LAST
   LOOP
      DBMS_OUTPUT.PUT_LINE(DEPTNO_TABLE(i));
   END LOOP;

END;
/
