
DECLARE
   TYPE Emp_ENAME    IS TABLE OF EMP.ENAME%TYPE INDEX BY BINARY_INTEGER;
   
   EMPNO_TABLE Emp_ENAME;
   SAL_TABLE EMP_SAL;
   HIREDATE_TABLE EMP_HIREDATE;
   
   CURSOR C_EMP IS
      SELECT ENAME, SAL, HIREDATE FROM EMP;
      
BEGIN
   OPEN C_EMP;
   LOOP
     FETCH C_EMP 
        BULK COLLECT INTO EMPNO_TABLE, SAL_TABLE, HIREDATE_TABLE 
        LIMIT 1000;
     EXIT WHEN EMPNO_TABLE.COUNT = 0;
--
--  Do some DML processing perhaps
--
     DBMS_OUTPUT.PUT_LINE(C_EMP%ROWCOUNT || ' Rows Fetched so far');

   END LOOP;
     
   CLOSE C_EMP;
   
END;
/
